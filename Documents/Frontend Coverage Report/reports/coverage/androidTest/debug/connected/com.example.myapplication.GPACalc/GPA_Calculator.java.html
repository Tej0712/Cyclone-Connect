package com.example.myapplication.GPACalc;

import android.os.Bundle;
import android.view.LayoutInflater;
import android.view.View;
import android.widget.ArrayAdapter;
import android.widget.Button;
import android.widget.EditText;
import android.widget.LinearLayout;
import android.widget.Spinner;
import android.widget.TextView;

import androidx.appcompat.app.AppCompatActivity;

import com.android.volley.Request;
import com.android.volley.toolbox.JsonObjectRequest;
import com.example.myapplication.R;
import com.example.myapplication.helper.VolleySingleton;

import org.json.JSONArray;
import org.json.JSONException;
import org.json.JSONObject;
public class GPA_Calculator extends AppCompatActivity {

    private LinearLayout coursesContainer;
    private Button btnAddCourse, btnCalculateGPA;
    private TextView tvResult;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.gpa_calc);

        coursesContainer = findViewById(R.id.coursesContainer);
        btnAddCourse = findViewById(R.id.btnAddCourse);
        btnCalculateGPA = findViewById(R.id.btnCalculateGPA);
        tvResult = findViewById(R.id.tvResult);

        btnAddCourse.setOnClickListener(v -> addCourseEntry());
        btnCalculateGPA.setOnClickListener(v -> calculateGPA());
    }

    private void addCourseEntry() {
        View courseEntry = LayoutInflater.from(this).inflate(R.layout.gpa_add_course, coursesContainer, false);
        Spinner spinnerGrade = courseEntry.findViewById(R.id.spinnerGrade);
        ArrayAdapter<CharSequence> adapter = ArrayAdapter.createFromResource(this,
                R.array.grades_array, android.R.layout.simple_spinner_item);
        adapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
        spinnerGrade.setAdapter(adapter);
        coursesContainer.addView(courseEntry);
    }

    private void calculateGPA() {
        JSONArray coursesArray = new JSONArray();

        for (int i = 0; i < coursesContainer.getChildCount(); i++) {
            View courseEntry = coursesContainer.getChildAt(i);
            String courseNumber = ((EditText) courseEntry.findViewById(R.id.editTextCourseNumber)).getText().toString();
            String credits = ((EditText) courseEntry.findViewById(R.id.editTextCredits)).getText().toString();
            String grade = ((Spinner) courseEntry.findViewById(R.id.spinnerGrade)).getSelectedItem().toString();

            JSONObject courseObject = new JSONObject();
            try {
                courseObject.put("course_number", courseNumber);
                courseObject.put("credits", credits);
                courseObject.put("grade", grade);
                coursesArray.put(courseObject);
            } catch (JSONException e) {
                e.printStackTrace();
            }
        }

        String url = "http://yourbackendapi.com/api/gpa";
        sendPostRequest(url, coursesArray);
    }

    private void sendPostRequest(String url, JSONArray requestBodyArray) {
        JSONObject requestBody = new JSONObject();
        try {
            requestBody.put("courses", requestBodyArray);  // Wrap the JSONArray inside a JSONObject with a key, e.g., "courses"
        } catch (JSONException e) {
            e.printStackTrace();  // Handle the exception here
        }

        JsonObjectRequest jsonObjectRequest = new JsonObjectRequest(Request.Method.POST, url, requestBody,
                response -> {
                    try {
                        // Assuming the response contains a key 'calculated_gpa' that holds the GPA value
                        String gpaResult = response.getString("calculated_gpa");
                        tvResult.setText("Your GPA is: " + gpaResult);
                        tvResult.setVisibility(View.VISIBLE);
                    } catch (JSONException e) {
                        e.printStackTrace();
                        tvResult.setText("Error in response");
                        tvResult.setVisibility(View.VISIBLE);
                    }
                },
                error -> {
                    error.printStackTrace();
                    tvResult.setText("Failed to calculate GPA");
                    tvResult.setVisibility(View.VISIBLE);
                });

        VolleySingleton.getInstance(this).addToRequestQueue(jsonObjectRequest);
    }

}
