public class ExerciseActivity extends AppCompatActivity implements ExerciseRVAdapter.ExerciseClickInterface {
    private static final String TAG = ExerciseActivity.class.getSimpleName();
    private static final String GET_EXERCISES_URL = "http://coms-309-046.class.las.iastate.edu:8080/exercises";

    private RecyclerView exerciseRV;
    private ArrayList<ExerciseRVModal> exerciseRVModalArrayList;
    private ExerciseRVAdapter exerciseRVAdapter;
    private Map<String, String> exerciseImageUrlMap;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_exercise);

        exerciseRV = findViewById(R.id.idRVExercise);
        exerciseRVModalArrayList = new ArrayList<>();
        exerciseRVAdapter = new ExerciseRVAdapter(exerciseRVModalArrayList, this);
        exerciseRV.setLayoutManager(new LinearLayoutManager(this));
        exerciseRV.setAdapter(exerciseRVAdapter);

        exerciseImageUrlMap = new HashMap<>();
        populateExerciseImageUrlMap();

        getExercisesFromBackend();
    }

    private void populateExerciseImageUrlMap() {
        exerciseImageUrlMap.put("Side Plank", "https://lottie.host/ccf1a3b2-bda3-455a-aefd-cc32f1a511fe/wT38vTq1Dr.json");
        exerciseImageUrlMap.put("Lunges", "https://lottie.host/7f79ef0c-3ce1-41c8-843f-35d6bd0ce77f/dgIQMbafF6.json");
        exerciseImageUrlMap.put("High Stepping", "https://lottie.host/cbf039b5-03f5-483d-bdae-892a6cf08ec5/XqGALV6c3E.json");
        exerciseImageUrlMap.put("Abs Crunches", "https://lottie.host/6a692e81-2cc2-4767-9c69-83081479cd2a/7Ql2DebrX8.json");
        exerciseImageUrlMap.put("Push Ups", "https://lottie.host/92f53510-5f96-4b55-b46f-36a8e5e9db8b/nIGtor0JGU.json");
    }

    private void getExercisesFromBackend() {
        VolleySingleton.getInstance(this).sendGetRequest(GET_EXERCISES_URL, new Response.Listener<JSONObject>() {
            @Override
            public void onResponse(JSONObject response) {
                try {
                    JSONArray exercisesArray = response.getJSONArray("exercises");
                    for (int i = 0; i < exercisesArray.length(); i++) {
                        JSONObject exerciseObj = exercisesArray.getJSONObject(i);
                        int exerciseId = exerciseObj.getInt("exerciseId");
                        String exerciseName = exerciseObj.getString("exercise_name");
                        int time = exerciseObj.getInt("time");

                        // Get the image URL and exercise description from the frontend map
                        String imageUrl = exerciseImageUrlMap.get(exerciseName);
                        String exerciseDescription = getExerciseDescription(exerciseName);

                        // Create an ExerciseRVModal object with the retrieved data
                        ExerciseRVModal exerciseRVModal = new ExerciseRVModal(
                                exerciseName,
                                exerciseDescription,
                                imageUrl,
                                0,
                                time
                        );
                        exerciseRVModalArrayList.add(exerciseRVModal);
                    }
                    exerciseRVAdapter.notifyDataSetChanged();
                } catch (JSONException e) {
                    Log.e(TAG, "Error parsing JSON response", e);
                }
            }
        }, new Response.ErrorListener() {
            @Override
            public void onErrorResponse(VolleyError error) {
                Log.e(TAG, "Error fetching exercises from backend", error);
            }
        });
    }

    private String getExerciseDescription(String exerciseName) {
        switch (exerciseName) {
            case "Side Plank":
                return getResources().getString(R.string.side_plank);
            case "Lunges":
                return getResources().getString(R.string.lunges);
            case "High Stepping":
                return getResources().getString(R.string.stepping);
            case "Abs Crunches":
                return getResources().getString(R.string.abs_crunches);
            case "Push Ups":
                return getResources().getString(R.string.push_ups);
            default:
                return "";
        }
    }

    @Override
    public void onExerciseClick(int position) {
        Intent i = new Intent(ExerciseActivity.this, ExerciseDetailActivity.class);
        i.putExtra("exerciseName", exerciseRVModalArrayList.get(position).getExerciseName());
        i.putExtra("imgUrl", exerciseRVModalArrayList.get(position).getImgUrl());
        i.putExtra("time", exerciseRVModalArrayList.get(position).getTime());
        i.putExtra("calories", exerciseRVModalArrayList.get(position).getCalories());
        i.putExtra("desc", exerciseRVModalArrayList.get(position).getExerciseDescription());
        startActivity(i);
    }
}
