package com.example.myapplication.CyFind;

import android.os.Bundle;
import android.widget.Button;
import android.widget.ImageView;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;
import com.android.volley.Request;
import com.android.volley.Response;
import com.android.volley.VolleyError;
import com.android.volley.toolbox.JsonObjectRequest;
import com.example.myapplication.R;
import com.example.myapplication.helper.VolleySingleton;
import org.json.JSONObject;

public class ItemDetails extends AppCompatActivity {

    private TextView title, address, dateFound, mail, description, category;
    private ImageView img;
    private Button backBtn, callBtn, smsBtn;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.itemdetails);

        // Initialize UI components
        title = findViewById(R.id.title);
        address = findViewById(R.id.address);
        dateFound = findViewById(R.id.dateFound);
        mail = findViewById(R.id.mail);
        description = findViewById(R.id.description);
        category = findViewById(R.id.category);
        img = findViewById(R.id.img);
        backBtn = findViewById(R.id.backBtn);
        callBtn = findViewById(R.id.call);
        smsBtn = findViewById(R.id.sms);

        // Set listeners (if any specific actions are needed for buttons)
        backBtn.setOnClickListener(v -> finish());
        // For call and sms buttons, intents can be set to perform actual actions.

        // Fetch item details from the backend
        long itemId = 123; // This should be retrieved via intent or another source.
        fetchItemDetails("http://yourbackend.com/api/items/" + itemId);
    }

    private void fetchItemDetails(String url) {
        JsonObjectRequest jsonObjectRequest = new JsonObjectRequest(Request.Method.GET, url, null,
                new Response.Listener<JSONObject>() {
                    @Override
                    public void onResponse(JSONObject response) {
                        // Parsing the response and updating UI
                        updateUI(response);
                    }
                }, new Response.ErrorListener() {
            @Override
            public void onErrorResponse(VolleyError error) {
                // Handle errors here, show a toast or log them
                error.printStackTrace();
            }
        });

        // Add the request to the VolleySingleton queue
        VolleySingleton.getInstance(this).addToRequestQueue(jsonObjectRequest);
    }

    private void updateUI(JSONObject item) {
        try {
            // Assuming JSON object response keys are consistent with TextView IDs
            title.setText(item.optString("title"));
            address.setText(item.optString("description")); // Assuming 'address' is for the description
            dateFound.setText(item.optString("dateFound"));
            mail.setText(item.optString("email"));
            description.setText(item.optString("description"));
            category.setText(item.optString("category"));

            // Additional fields can be set as needed
            // Image handling needs to be implemented if necessary
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
