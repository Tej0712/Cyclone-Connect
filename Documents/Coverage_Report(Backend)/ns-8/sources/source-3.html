 package onetoone.HealthAndWellness;
 
 import org.springframework.beans.factory.annotation.Autowired;
 import org.springframework.http.HttpStatus;
 import org.springframework.http.ResponseEntity;
 import org.springframework.web.bind.annotation.*;
 
 import java.time.Duration;
 import java.util.List;
 import java.util.stream.Collectors;
 
 @RestController
 @RequestMapping("/exerciseBenefits")
 public class ExerciseBenefitsController {
 
     private final ExerciseBenefitsRepository exerciseBenefitsRepository;
     private final ExerciseRepository exerciseRepository;
 
     @Autowired
     public ExerciseBenefitsController(ExerciseBenefitsRepository exerciseBenefitsRepository, ExerciseRepository exerciseRepository) {
         this.exerciseBenefitsRepository = exerciseBenefitsRepository;
         this.exerciseRepository = exerciseRepository;
     }
 
     // POST method for creating a new ExerciseBenefits entry
     @PostMapping
     public ResponseEntity<ExerciseBenefitsDTO> createExerciseBenefits(@RequestBody ExerciseBenefitsDTO exerciseBenefitsDTO) {
         return exerciseRepository.findById(exerciseBenefitsDTO.getExerciseId())
                 .map(exercise -> {
                     ExerciseBenefits exerciseBenefits = new ExerciseBenefits();
                     // Set properties from DTO to entity
                     exerciseBenefits.setCaloriesBurned(exerciseBenefitsDTO.getCaloriesBurned());
                     exerciseBenefits.setExerciseDuration(Duration.ofMinutes(exerciseBenefitsDTO.getExerciseDurationMinutes()));
                     exerciseBenefits.setBenefits(exerciseBenefitsDTO.getBenefits());
                     exerciseBenefits.setExercise(exercise); // Linking with Exercise
                     exerciseBenefits = exerciseBenefitsRepository.save(exerciseBenefits);
                     return new ResponseEntity<>(ExerciseBenefitsDTO.fromEntity(exerciseBenefits), HttpStatus.CREATED);
                 })
                 .orElseGet(() -> ResponseEntity.notFound().build());
     }
 
     // GET method for retrieving all ExerciseBenefits entries
     @GetMapping
     public ResponseEntity<List<ExerciseBenefitsDTO>> getAllExerciseBenefits() {
         List<ExerciseBenefitsDTO> exerciseBenefitsDTOs = exerciseBenefitsRepository.findAll().stream()
                 .map(ExerciseBenefitsDTO::fromEntity)
                 .collect(Collectors.toList());
         return ResponseEntity.ok(exerciseBenefitsDTOs);
     }
 
     // GET method for retrieving a single ExerciseBenefits entry by id
     @GetMapping("/{id}")
     public ResponseEntity<ExerciseBenefitsDTO> getExerciseBenefitsById(@PathVariable Long id) {
         return exerciseBenefitsRepository.findById(id)
                 .map(exerciseBenefits -> ResponseEntity.ok(ExerciseBenefitsDTO.fromEntity(exerciseBenefits)))
                 .orElseGet(() -> ResponseEntity.notFound().build());
     }
 
     // PUT method for updating an existing ExerciseBenefits entry
     @PutMapping("/{id}")
     public ResponseEntity<ExerciseBenefitsDTO> updateExerciseBenefits(@PathVariable Long id, @RequestBody ExerciseBenefitsDTO exerciseBenefitsDTO) {
         return exerciseBenefitsRepository.findById(id)
                 .map(existingExerciseBenefits -> {
                     // Update properties from DTO to entity
                     existingExerciseBenefits.setCaloriesBurned(exerciseBenefitsDTO.getCaloriesBurned());
                     existingExerciseBenefits.setExerciseDuration(Duration.ofMinutes(exerciseBenefitsDTO.getExerciseDurationMinutes()));
                     existingExerciseBenefits.setBenefits(exerciseBenefitsDTO.getBenefits());
                     // Update the ExerciseBenefits entry
                     existingExerciseBenefits = exerciseBenefitsRepository.save(existingExerciseBenefits);
                     return ResponseEntity.ok(ExerciseBenefitsDTO.fromEntity(existingExerciseBenefits));
                 })
                 .orElseGet(() -> ResponseEntity.notFound().build());
     }
 
     // DELETE method for deleting an ExerciseBenefits entry
     @DeleteMapping("/{id}")
     public ResponseEntity<Void> deleteExerciseBenefits(@PathVariable Long id) {
         if (!exerciseBenefitsRepository.existsById(id)) {
             return ResponseEntity.notFound().build();
         }
         exerciseBenefitsRepository.deleteById(id);
         return ResponseEntity.noContent().build();
     }
 }
